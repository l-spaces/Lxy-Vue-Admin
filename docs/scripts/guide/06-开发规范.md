## 1. 代码规范

### 1.1 TypeScript 规范

#### 类型定义

```typescript
// 使用 interface 定义对象类型
interface UserInfo {
  userId: string;
  userName: string;
  email: string;
}

// 使用 type 定义联合类型、元组等
type Status = 'active' | 'inactive';
type Point = [number, number];

// 避免使用 any，使用 unknown 代替
function processData(data: unknown) {
  if (typeof data === 'string') {
    // ...
  }
}
```

#### 命名规范

```typescript
// 接口使用 PascalCase
interface UserInfo { ... }

// 类型别名使用 PascalCase
type UserRole = 'admin' | 'user';

// 变量使用 camelCase
const userName = 'test';
const userList = [];

// 常量使用 UPPER_SNAKE_CASE
const MAX_RETRY_COUNT = 3;
const API_BASE_URL = '/api';

// 函数使用 camelCase
function getUserInfo() { ... }

// 组件使用 PascalCase
const UserTable = defineComponent({ ... });

// 文件名使用 kebab-case
// user-info.ts
// user-table.vue
```

#### 枚举使用

```typescript
// 使用 const enum 或 as const
const enum ApiStatus {
  Success = 200,
  Unauthorized = 401,
  Forbidden = 403,
  ServerError = 500,
}

// 或使用对象 + as const
const ApiCode = {
  Success: 200,
  Unauthorized: 401,
} as const;
```

### 1.2 Vue 组件规范

#### 组件结构

```vue
<script setup lang="ts">
// 1. imports
import { ref, computed, onMounted } from 'vue';
import type { PropType } from 'vue';

// 2. 类型定义
interface Props {
  title?: string;
  data?: User[];
}

interface Emits {
  (e: 'update', value: string): void;
  (e: 'delete', id: string): void;
}

// 3. Props 定义
const props = withDefaults(defineProps<Props>(), {
  title: '默认标题',
  data: () => [],
});

// 4. Emits 定义
const emit = defineEmits<Emits>();

// 5. 响应式数据
const loading = ref(false);
const selectedKeys = ref<string[]>([]);

// 6. 计算属性
const filteredData = computed(() => {
  return props.data.filter(item => item.status === 'active');
});

// 7. 方法
function handleUpdate(value: string) {
  emit('update', value);
}

// 8. 生命周期
onMounted(() => {
  fetchData();
});
</script>

<template>
  <div class="component-container">
    <h2>{{ title }}</h2>
    <slot />
  </div>
</template>

<style scoped>
.component-container {
  padding: 16px;
}
</style>
```

#### 组件命名

```typescript
// 组件文件使用 PascalCase 或 kebab-case
// UserInfo.vue 或 user-info.vue

// 注册组件时使用 PascalCase
import UserInfo from './UserInfo.vue';

app.component('UserInfo', UserInfo);
```

#### Props 验证

```typescript
// 推荐：使用 TypeScript 类型
```

## 2. 组件开发规范

### 2.1 组件设计原则

#### 单一职责

每个组件只负责一个功能：

```vue
<!-- 推荐：职责单一 -->
<template>
  <div class="user-avatar">
    <img :src="avatar" :alt="name" />
  </div>
</template>

<!-- 不推荐：职责过多 -->
<template>
  <div class="user-card">
    <img :src="avatar" />
    <h3>{{ name }}</h3>
    <p>{{ email }}</p>
    <button @click="edit">编辑</button>
    <button @click="delete">删除</button>
  </div>
</template>
```

#### 可复用性

组件应该可以在多个场景复用：

```typescript
// 推荐：通过 props 控制行为
interface Props {
  mode: 'view' | 'edit';
  data: User;
  showActions?: boolean;
}

// 不推荐：硬编码行为
const mode = 'edit'; // 无法复用
```

### 2.2 Props 定义

```typescript
// 推荐：使用 TypeScript 接口
interface Props {
  // 必填属性
  title: string;
  
  // 可选属性
  size?: 'small' | 'medium' | 'large';
  
  // 带默认值的属性
  disabled?: boolean;
  
  // 数组/对象类型
  options?: SelectOption[];
  
  // 函数类型
  onChange?: (value: string) => void;
}

const props = withDefaults(defineProps<Props>(), {
  size: 'medium',
  disabled: false,
  options: () => [],
});
```

### 2.3 Events 定义

```typescript
// 推荐：使用类型化 Emits
interface Emits {
  (e: 'update:modelValue', value: string): void;
  (e: 'change', value: string, oldValue: string): void;
  (e: 'submit', data: FormData): void;
}

const emit = defineEmits<Emits>();

// 使用
emit('update:modelValue', newValue);
emit('change', newValue, oldValue);
emit('submit', formData);
```

### 2.4 Slots 定义

```vue
<script setup lang="ts">
interface Props {
  title?: string;
}

interface Slots {
  default(): any;
  header(props: { title: string }): any;
  footer(): any;
}

defineSlots<Slots>();
</script>

<template>
  <div class="card">
    <div class="card-header">
      <slot name="header" :title="title">
        <h3>{{ title }}</h3>
      </slot>
    </div>
    
    <div class="card-body">
      <slot />
    </div>
    
    <div class="card-footer">
      <slot name="footer" />
    </div>
  </div>
</template>
```

## 3. API 开发规范

### 3.1 API 分层

```
api/
├── system/              # 模块目录
│   ├── user/
│   │   ├── index.ts     # API 方法
│   │   └── model.d.ts   # 类型定义
```

**API 文件结构**：

```typescript
// api/system/user/index.ts

// 1. 导入类型
import type { PageQuery, PageResult } from '#/api/common';

// 2. 导入请求实例
import { alovaInstance } from '#/utils/http';

// 3. 定义 API 枚举
enum Api {
  userList = '/system/user/list',
  userDetail = '/system/user/{userId}',
  userAdd = '/system/user',
  userUpdate = '/system/user',
  userDelete = '/system/user',
}

// 4. 导出接口方法
export async function userList(query: UserQuery) {
  return alovaInstance.get<PageResult<User>>(Api.userList, {
    params: query,
  });
}

export async function userDetail(userId: string) {
  return alovaInstance.get<User>(
    Api.userDetail.replace('{userId}', userId)
  );
}
```

### 3.2 错误处理

```typescript
// 推荐：统一错误处理
async function fetchData() {
  try {
    loading.value = true;
    const data = await userList(query);
    return data;
  } catch (error) {
    // 错误已在拦截器中处理
    console.error('获取用户列表失败:', error);
    return { rows: [], total: 0 };
  } finally {
    loading.value = false;
  }
}

// 不推荐：忽略错误
async function fetchData() {
  const data = await userList(query); // 可能未捕获错误
  return data;
}
```

## 4. 状态管理规范

### 4.1 Store 定义

```typescript
// store/user.ts
import { defineStore } from 'pinia';
import { ref, computed } from 'vue';

export const useUserStore = defineStore('user', () => {
  // State
  const userInfo = ref<UserInfo | null>(null);
  const permissions = ref<string[]>([]);

  // Getters
  const isLoggedIn = computed(() => !!userInfo.value);
  const userName = computed(() => userInfo.value?.nickName ?? '');

  // Actions
  function setUserInfo(info: UserInfo) {
    userInfo.value = info;
  }

  function setPermissions(perms: string[]) {
    permissions.value = perms;
  }

  function $reset() {
    userInfo.value = null;
    permissions.value = [];
  }

  return {
    // State
    userInfo,
    permissions,
    // Getters
    isLoggedIn,
    userName,
    // Actions
    setUserInfo,
    setPermissions,
    $reset,
  };
});
```

### 4.2 Store 使用

```typescript
// 在组件中使用
import { useUserStore } from '#/store';

const userStore = useUserStore();

// 访问 State
console.log(userStore.userInfo);

// 访问 Getters
console.log(userStore.isLoggedIn);

// 调用 Actions
userStore.setUserInfo(info);

// 重置 Store
userStore.$reset();
```

## 5. Git 提交规范

### 5.1 分支管理

| 分支类型 | 命名规则 | 说明 |
|----------|----------|------|
| 主分支 | `master` | 生产环境代码 |
| 开发分支 | `develop` | 开发环境代码 |
| 功能分支 | `feature/xxx` | 新功能开发 |
| 修复分支 | `bugfix/xxx` | Bug 修复 |
| 发布分支 | `release/x.x.x` | 版本发布 |
| 热修复分支 | `hotfix/xxx` | 紧急修复 |

### 5.2 Commit 格式

```
<type>(<scope>): <subject>

<body>

<footer>
```

**Type 类型**：

| 类型 | 说明 | 示例 |
|------|------|------|
| `feat` | 新功能 | feat(user): add user import feature |
| `fix` | Bug 修复 | fix(auth): fix token refresh issue |
| `docs` | 文档更新 | docs: update README |
| `style` | 代码格式 | style: format code |
| `refactor` | 重构 | refactor(api): restructure user API |
| `perf` | 性能优化 | perf: optimize table rendering |
| `test` | 测试 | test: add user store tests |
| `chore` | 构建/工具 | chore: update dependencies |
| `ci` | CI 配置 | ci: add GitHub Actions |
| `build` | 打包构建 | build: update vite config |

**示例**：

```bash
# 新功能
feat(user): 添加用户导入功能

- 支持 Excel 导入
- 支持 CSV 导入
- 添加导入模板下载

Closes #123

# Bug 修复
fix(auth): 修复登录超时无法跳转的问题

修复了在登录超时后，页面无法正确跳转到登录页的问题。

# 重构
refactor(api): 重构用户管理 API 结构

将用户相关的 API 接口统一迁移到 api/system/user 目录下。
```

### 6.2 代码复用

#### Composables

```typescript
// composables/useTable.ts
export function useTable<T>(api: (query: any) => Promise<PageResult<T>>) {
  const loading = ref(false);
  const data = ref<T[]>([]);
  const total = ref(0);
  const query = ref({ pageNum: 1, pageSize: 10 });

  async function fetchData() {
    loading.value = true;
    try {
      const result = await api(query.value);
      data.value = result.rows;
      total.value = result.total;
    } finally {
      loading.value = false;
    }
  }

  return { loading, data, total, query, fetchData };
}

// 使用
const { loading, data, fetchData } = useTable(userList);
```

#### 工具函数

```typescript
// utils/file/index.ts
export function downloadFile(url: string, filename: string) {
  const link = document.createElement('a');
  link.href = url;
  link.download = filename;
  link.click();
}

export function formatFileSize(bytes: number): string {
  if (bytes === 0) return '0 B';
  const k = 1024;
  const sizes = ['B', 'KB', 'MB', 'GB', 'TB'];
  const i = Math.floor(Math.log(bytes) / Math.log(k));
  return `${parseFloat((bytes / Math.pow(k, i)).toFixed(2))} ${sizes[i]}`;
}
```
